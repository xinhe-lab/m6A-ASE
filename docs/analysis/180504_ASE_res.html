def RDS2df(RBP):
    df = readRDS(f'{path}/Data/ASE/samples/{RBP}.wasp_remapped.qual10.dedup.quasar.ase.rds')
    df = pandas2ri.ri2py(df)
    tmp = df[0][0]
    print (type(tmp[0]))
    RBP_df = pd.DataFrame(data = OrderedDict({"rsID": tmp[0], "chr": list(tmp[1]), "pos": list(tmp[2]), 
                                              "beta": list(tmp[3]), "se": list(tmp[4]), "pval": list(tmp[5])}))
    return RBP_df
